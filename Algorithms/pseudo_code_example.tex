\begin{algorithm}[!ht]
    \caption{Example of Objective Function - Maximize the Return} \label{algo:example_obj_func}
    \begin{algorithmic}[1]
        \STATE $Sum \Leftarrow 0$
        \FOR{each $Grouping$ in $L$}
            \STATE $Sum \Leftarrow Sum$ + value of $Grouping$
        \ENDFOR
        \RETURN $Sum$
    \end{algorithmic}
\end{algorithm}
